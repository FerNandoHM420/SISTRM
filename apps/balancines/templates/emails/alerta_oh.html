# apps/balancines/services/alertas_oh.py @classmethod def _enviar_email_inmediato(cls, alerta): """ Envía email INMEDIATAMENTE cuando se crea una alerta - VERSIÓN ROBUSTA """ try: from django.core.mail import send_mail from django.conf import settings balancin = alerta.balancin torre = balancin.torre # Mapeo simple de niveles if alerta.nivel == 'CRITICO': emoji = '🔴' asunto = f'{emoji} ALERTA CRÍTICA: {balancin.codigo}' destinatarios = ['miguefernandohuanca@gmail.com'] # Cambia por tus destinatarios elif alerta.nivel == 'ALERTA': emoji = '🟡' asunto = f'{emoji} ALERTA: {balancin.codigo}' destinatarios = ['miguefernandohuanca@gmail.com'] elif alerta.nivel == 'VENCIDO': emoji = '🟠' asunto = f'{emoji} ALERTA VENCIDA: {balancin.codigo}' destinatarios = ['miguefernandohuanca@gmail.com'] else: emoji = '⚠️' asunto = f'{emoji} Alerta: {balancin.codigo}' destinatarios = ['miguefernandohuanca@gmail.com'] # Mensaje en texto plano (más confiable que HTML) mensaje = f""" {alerta.nivel} - {balancin.codigo} {'=' * 40} 📍 Ubicación: {torre.linea.nombre if torre else 'N/A'} - Torre {torre.numero_torre if torre else 'N/A'} ⬆️ Sentido: {balancin.get_sentido_display()} 📊 Backlog: {alerta.backlog_momento} horas ⏱️ Horas operación: {alerta.horas_operacion_momento} horas 📅 Rango OH: {balancin.rango_horas_cambio_oh} horas 🔗 Ver detalles: {settings.SITE_URL}/dashboard-alertas/ {'=' * 40} Este es un mensaje automático del Sistema TRM Fecha: {timezone.now().strftime('%d/%m/%Y %H:%M:%S')} """ # Envío con fail_silently=False para ver errores send_mail( subject=asunto, message=mensaje, from_email=settings.DEFAULT_FROM_EMAIL, recipient_list=destinatarios, fail_silently=False, # IMPORTANTE: False para ver errores ) print(f"✅ Email enviado a {destinatarios} para {balancin.codigo}") return True except Exception as e: print(f"❌ Error enviando email: {e}") import traceback traceback.print_exc() return False
